
\begin{grammar}
	<Stmt> ::= `assume' <Expr> ;
	\alt $v$ `:=' <Expr> ;
	\alt `havoc' $v$ ;
	\alt `while' ( <WildcardExpr> ) <Stmt>* ;
	\alt `if' ( <WildcardExpr> ) <Stmt>* `else' <Stmt>*
	
	<Expr> ::= <Expr> <BinOp> <Expr>
	\alt <Var>
	\alt <UnOp> <Expr>
	\alt `true'
	\alt `false'
	
	<WildCardExpr> ::= <Expr> | `*'
	
	<Var> ::= 	$v$ | <Const>

	<Const> ::= $c$
	
	<BinOp> ::= `+' | `-' | `*' | `/' | `\%'
	\alt `\&\&' | `||' | `==' | `!='
	\alt `<' | `<=' | `>' | `>='
	
	<UnOp> ::= `-' | `!'
\end{grammar}
