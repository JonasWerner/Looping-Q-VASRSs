	\newcommand{\up}{\ensuremath{\begin{bmatrix} x \\ y \\ z \end{bmatrix}}}
	\begin{definition}
		Given a transition formula $F$, a \qvasr-abstraction for $F$ is a tuple $(S_F, V_F)$ consisting of a matrix $S_F$ and a \qvasr\ $V_F$. The matrix $S_F$ acts as a linear transformation of $F$ to $V_F$. Meaning, for every transition \\
		$
		\begin{bmatrix}
			x_1 \\
			\vdots \\
			x_n
		\end{bmatrix}
		\rightarrow_F
		\begin{bmatrix}
			x_1' \\
			\vdots \\
			x_n'
		\end{bmatrix}
		$
		in $F$, there is: \ \
		$ S_F \cdot
		\begin{bmatrix}
			x_1' \\
			\vdots \\
			x_n'
		\end{bmatrix}
		=
		S_F \cdot
		\begin{bmatrix}
			x_1 \\
			\vdots \\
			x_n
		\end{bmatrix}
		*
		%%%%%%%%%%%%%%%%%%%% RESET
		\only<3-5>{
			\begin{bmatrix}
				\color{red}{0} \\
				\color{red}{\vdots} \\
				\color{red}{0}
			\end{bmatrix}	
		}
		%%%%%%%%%%%%%%%%%%%%% VANILLA
		\only<1-2>{
			\begin{bmatrix}
				r_1 \\
				\vdots \\
				r_n
			\end{bmatrix}
		}
		%%%%%%%%%%%%%%%%%%%%%% NOT RESET
		\only<6->{
			\begin{bmatrix}
				\color{red}{1} \\
				\color{red}{\vdots} \\
				\color{red}{1}
			\end{bmatrix}
		}
		+
		\begin{bmatrix}
			a_1 \\
			\vdots \\
			a_n
		\end{bmatrix}	
		$
		in $V_F$
	\end{definition}
